```typescript
import { FC } from 'react'

interface Props {
  firstName: string
  lastName: string
  age: number
  hobbies: string[]
}

const SomeCoponent: FC<Props> = ({ firstName, lastName, age, hobbies }) => {
  return (
    <div>
      <h4>
        Full name: {firstName} {lastName}
      </h4>
      <h5>Age: {age}</h5>
      <h6>
        Hobbies:
        {hobbies.map((hobby: string) => (
          <span>{hobby}</span>
        ))}
      </h6>
    </div>
  )
}

export default SomeComponent
```
